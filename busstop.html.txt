    <script>
        // 白虹樓 (往荃灣方向) 喺九巴 API 嘅 Stop ID 係 963CC709F70B0E4E
        const STOP_ID = '963CC709F70B0E4E';
        const ROUTES = ['290', '290A', '290X'];


        async function fetchETA() {
            try {
                // 更新時間戳記
                const now = new Date();
                document.getElementById('update-time').innerText = `更新於: ${now.toTimeString().split(' ')[0]}`;


                for (const route of ROUTES) {
                    // Call 政府 Data.gov.hk 嘅九巴 API
                    const response = await fetch(`https://data.etabus.gov.hk/v1/transport/kmb/stop-eta/${STOP_ID}`);
                    const data = await response.json();
                    
                    // 篩選出對應路線、往荃灣西方向 (Inbound/Outbound 視乎九巴定義，通常比對目的地)
                    // 290系列去荃灣西係屬於 OUTBOUND
                    const etas = data.data.filter(item => item.route === route && item.dir === 'O');


                    const etaElement = document.getElementById(`eta-${route}`);
                    
                    if (etas.length > 0 && etas[0].eta) {
                        // 計算剩餘分鐘
                        const etaTime = new Date(etas[0].eta);
                        const diffMs = etaTime - now;
                        const diffMins = Math.max(0, Math.floor(diffMs / 1000 / 60));
                        
                        if (diffMins === 0) {
                            etaElement.innerText = '即將抵達';
                            etaElement.className = 'eta-time text-danger animate-pulse'; // 轉紅色提示
                        } else {
                            etaElement.innerText = `${diffMins} 分鐘`;
                            etaElement.className = 'eta-time text-success';
                        }
                    } else {
                        etaElement.innerText = '尾班車已過';
                        etaElement.className = 'eta-time text-muted';
                    }
                }
            } catch (error) {
                console.error('獲取巴士數據出錯:', error);
            }
        }


        // 第一次載入
        fetchETA();


        // 每 30 秒自動更新一次 (30000 毫秒)
        setInterval(fetchETA, 30000);
    </script>
